697inline LogOutputFunctionCB GetLogOutputFunction()
698{
699 using Wrapper = UniqueWrapper<LogOutputFunctionCB>;
700 LogOutputFunction cb;
701 void* userdata;
702 if (userdata == nullptr) {
703 return [cb](LogCategory c, LogPriority p, StringParam m) {
704 cb(nullptr, c, p, m);
705 };
706 }
707 GetLogOutputFunction(&cb, &userdata);
708 if (!Wrapper::contains(userdata)) {
709 return [cb, userdata](LogCategory c, LogPriority p, StringParam m) {
710 cb(userdata, c, p, m);
711 };
712 }
713 return Wrapper::at(userdata);
714}
