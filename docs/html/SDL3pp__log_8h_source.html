705inline LogOutputFunctionCB GetLogOutputFunction()
706{
707 using Wrapper = UniqueWrapper<LogOutputFunctionCB>;
708 LogOutputFunction cb;
709 void* userdata;
710 if (userdata == nullptr) {
711 return [cb](LogCategory c, LogPriority p, StringParam m) {
712 cb(nullptr, c, p, m);
713 };
714 }
715 GetLogOutputFunction(&cb, &userdata);
716 if (!Wrapper::contains(userdata)) {
717 return [cb, userdata](LogCategory c, LogPriority p, StringParam m) {
718 cb(userdata, c, p, m);
719 };
720 }
721 return Wrapper::at(userdata);
722}
