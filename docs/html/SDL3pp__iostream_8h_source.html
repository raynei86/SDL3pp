256 std::string Read(size_t size = -1)
257 {
258 Sint64 pos = Tell();
259 auto curSize = SDL_GetIOSize(get());
260 if ((curSize < 0 || pos < 0)) {
261 if (size == size_t(-1)) return {};
262 } else if (curSize - pos <= 0) {
263 return {};
264 } else if (curSize - pos < size) {
265 size = curSize - pos;
266 }
267 std::string result(size, 0);
268 auto actualSize = Read(result);
269 if (actualSize < size) result.resize(actualSize);
270 return result;
271 }
