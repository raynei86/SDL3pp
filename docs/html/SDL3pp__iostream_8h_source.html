566 std::string Read(size_t size = -1)
567 {
568 Sint64 pos = Tell();
569 auto curSize = SDL_GetIOSize(get());
570 if ((curSize < 0 || pos < 0)) {
571 if (size == size_t(-1)) return {};
572 } else if (curSize - pos <= 0) {
573 return {};
574 } else if (curSize - pos < size) {
575 size = curSize - pos;
576 }
577 std::string result(size, 0);
578 auto actualSize = Read(result);
579 if (actualSize < size) result.resize(actualSize);
580 return result;
581 }
