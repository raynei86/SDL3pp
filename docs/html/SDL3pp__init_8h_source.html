903inline bool SDL::updateActive(bool active)
904{
905 static std::atomic_bool currentlyInitd{false};
906 bool result = !currentlyInitd.exchange(active);
907 if (active && !result) {
908 SetErrorUnformatted(
909 "Can not initialize, there is already an active instance");
910 } else
911 m_active = active;
912 return result;
913}
